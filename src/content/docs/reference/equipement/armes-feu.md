Les armes à feu utilisent la compétence **Armes à feu**. Le Code de dégâts correspond à la puissance du calibre — deux armes chambrées pour le même calibre ont le même Code de base. La **Pénétration** réduit la Résilience de la cible pour cette attaque.

**Modes de tir :** SA = semi-automatique &middot; DA/SA = double/simple action &middot; FA = full-auto (rafale/saturation) &middot; Répétition manuelle = un coup par action, rechargement inclus.

function renderFeu() {
  var search = document.getElementById('search-feu').value.toLowerCase();
  var type = document.getElementById('filter-type').value;
  var version = document.getElementById('filter-version').value;
  var filtered = DATA_FEU.filter(function(r) {
    return (!search || r["Nom de l'arme"].toLowerCase().includes(search) || r["Calibre"].toLowerCase().includes(search))
      && (!type || r["Type"] === type)
      && (!version || r["Version"] === version);
  });
  var tbody = document.getElementById('tbody-feu');
  tbody.innerHTML = filtered.map(function(r) {
    return '<tr style="border-bottom:0.5px solid var(--color-border-tertiary);">'
      + '<td style="padding:7px 10px;font-weight:500;color:var(--color-text-primary);">' + r["Nom de l'arme"] + '</td>'
      + '<td style="padding:7px 10px;color:var(--color-text-secondary);">' + r["Type"] + '</td>'
      + '<td style="padding:7px 10px;color:var(--color-text-secondary);">' + r["Calibre"] + '</td>'
      + '<td style="padding:7px 10px;font-family:monospace;font-size:12px;">' + r["Mode de tir"] + '</td>'
      + '<td style="padding:7px 10px;text-align:center;">' + r["Capacité"] + '</td>'
      + '<td style="padding:7px 10px;text-align:center;font-weight:500;color:#ff8c42;">' + r["Dégats"] + '</td>'
      + '<td style="padding:7px 10px;text-align:center;color:#a080ff;">' + r["Pénétration"] + '</td>'
      + '</tr>';
  }).join('');
  document.getElementById('count-feu').textContent = filtered.length + ' arme(s) affichée(s)';
}

var sortDir = {};
function sortTable(id, col) {
  var keys = ["Nom de l'arme","Type","Calibre","Mode de tir","Capacité","Dégats","Pénétration"];
  var key = keys[col];
  sortDir[col] = !sortDir[col];
  DATA_FEU.sort(function(a,b) {
    var va = a[key], vb = b[key];
    if (typeof va === 'number' && typeof vb === 'number') return sortDir[col] ? va-vb : vb-va;
    return sortDir[col] ? String(va).localeCompare(String(vb)) : String(vb).localeCompare(String(va));
  });
  renderFeu();
}

document.getElementById('search-feu').addEventListener('input', renderFeu);
